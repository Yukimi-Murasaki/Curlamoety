CuriosJSEvents.registerRenderer(event=>{
    event.register(
        'curlamoety:elemental_gloves',
        context => {
            let {
                stack,
                slotContext,
                matrixStack,
                renderLayerParent,
                renderTypeBuffer,
                light,
                limbSwing,
                limbSwingAmount,
                partialTicks,
                ageInTicks,
                netHeadYaw,
                headPitch
            } = context
            let { modelManager } = Client
            let entity = slotContext.entity()
            let model = modelManager.getModel(new ModelResourceLocation(stack.id, "curlamoety:models/curios/elemental_gloves.json"))
            matrixStack.pushPose()
            CuriosRenderer.translateIfSneaking(matrixStack, entity)
            matrixStack.mulPose(RotationAxis.YP.deg(limbSwing))
            matrixStack.mulPose(RotationAxis.XP.deg(limbSwingAmount))
            Client.itemRenderer.render(
                stack,
                'third_person_right_hand',
                false,
                matrixStack,
                renderTypeBuffer,
                light,
                OverlayTexture.NO_OVERLAY,
                model
            )
            matrixStack.popPose()
        }
    )
})

ItemEvents.entityInteracted('stick',event=>{
    let {player,target} = event
    if(target==null||!target.isLiving())return;
    let item1 = target.getItemBySlot("mainhand")
    let item2 = target.getItemBySlot("offhand")
    let item3 = target.getItemBySlot("head")
    let item4 = target.getItemBySlot("chest")
    let item5 = target.getItemBySlot("legs")
    let item6 = target.getItemBySlot("feet")
    if(item1 == null){
        player.tell("no")
    }else{
        player.tell(item1.displayName)
    }
    if(item2 == null){
        player.tell("no")
    }else{
        player.tell(item2.displayName)
    }
    if(item3 == null){

    }else{
        player.tell(item3.displayName)
    }
    if(item4 == null){
        player.tell("no")

        player.tell(item4.displayName)

        player.tell("no")
